384  /* Move all clock delays together */

387  /* Setup EMC Clock Divider for divide by 2 - this is done in both the CCU (clocking)

390  Chip_Clock_EnableOpts(CLK_MX_EMC_DIV, true, true, 2);

409 /* Set up and initialize clocking prior to call to main */
410 void Board_SetupClocking(void)
411 {
412  int i;
413 
414  Chip_SetupCoreClock(CLKIN_CRYSTAL, MAX_CLOCK_FREQ, true);
415 
416  /* Setup system base clocks and initial states. This won't enable and
417  disable individual clocks, but sets up the base clock sources for
418  each individual peripheral clock. */
419  for (i = 0; i < (sizeof(InitClkStates) / sizeof(InitClkStates[0])); i++) {
420  Chip_Clock_SetBaseClock(InitClkStates[i].clk, InitClkStates[i].clkin,
421  InitClkStates[i].autoblock_enab, InitClkStates[i].powerdn);
422  }
423 
424  /* Reset and enable 32Khz oscillator */
425  LPC_CREG->CREG0 &= ~((1 << 3) | (1 << 2));
426  LPC_CREG->CREG0 |= (1 << 1) | (1 << 0);
427 
428  /* Setup a divider E for main PLL clock switch SPIFI clock to that divider.
429  Divide rate is based on CPU speed and speed of SPI FLASH part. */
430 #if (MAX_CLOCK_FREQ > 180000000)
431  Chip_Clock_SetDivider(CLK_IDIV_E, CLKIN_MAINPLL, 5);
432 #else
433  Chip_Clock_SetDivider(CLK_IDIV_E, CLKIN_MAINPLL, 4);
434 #endif
435  Chip_Clock_SetBaseClock(CLK_BASE_SPIFI, CLKIN_IDIVE, true, false);
436 
437  /* Attach main PLL clock to divider C with a divider of 2 */
438  Chip_Clock_SetDivider(CLK_IDIV_C, CLKIN_MAINPLL, 2);
439 }
