170 STATIC INLINE void Chip_LCD_PowerOn(LPC_LCD_T *pLCD)
171 {
172  volatile int i;
173  pLCD->CTRL |= CLCDC_LCDCTRL_PWR;
174  for (i = 0; i < 1000000; i++) {}
