50 /* Setup the OUTPUT pin corresponding to the PWM index */
51 void Chip_SCTPWM_SetOutPin(LPC_SCT_T *pSCT, uint8_t index, uint8_t pin)
52 {
53  int ix = (int) index;
54  pSCT->EVENT[ix].CTRL = index | (1 << 12);
55  pSCT->EVENT[ix].STATE = 1;
56  pSCT->OUT[pin].SET = 1;
57  pSCT->OUT[pin].CLR = 1 << ix;
58 
59  /* Clear the output in-case of conflict */
60  pSCT->RES = (pSCT->RES & ~(3 << (pin << 1))) | (0x01 << (pin << 1));
61 
62  /* Set and Clear do not depend on direction */
63  pSCT->OUTPUTDIRCTRL = (pSCT->OUTPUTDIRCTRL & ~(3 << (pin << 1)));
64 }

66 /* Set the PWM frequency */
67 void Chip_SCTPWM_SetRate(LPC_SCT_T *pSCT, uint32_t freq)
68 {
69  uint32_t rate;
70 
71  rate = Chip_Clock_GetRate(CLK_MX_SCT) / freq;;
72 
73  /* Stop the SCT before configuration */
74  Chip_SCTPWM_Stop(pSCT);
75 
76  /* Set MATCH0 for max limit */
77  pSCT->REGMODE_L = 0;
78  pSCT->REGMODE_H = 0;
79  Chip_SCT_SetMatchCount(pSCT, SCT_MATCH_0, 0);
80  Chip_SCT_SetMatchReload(pSCT, SCT_MATCH_0, rate);
81  pSCT->EVENT[0].CTRL = 1 << 12;
82  pSCT->EVENT[0].STATE = 1;
83  pSCT->LIMIT_L = 1;
84 
85  /* Set SCT Counter to count 32-bits and reset to 0 after reaching MATCH0 */
86  Chip_SCT_Config(pSCT, SCT_CONFIG_32BIT_COUNTER | SCT_CONFIG_AUTOLIMIT_L);
87 }
