80 /* Setup Chip Core clock */
81 void Chip_SetupCoreClock(CHIP_CGU_CLKIN_T clkin, uint32_t core_freq, bool setbase)
82 {
83  int i;
84  if (clkin == CLKIN_CRYSTAL) {
85  /* Switch main system clocking to crystal */
86  Chip_Clock_EnableCrystal();
87  }
88  Chip_Clock_SetBaseClock(CLK_BASE_MX, clkin, true, false);
89 
90  if (core_freq > 110000000UL) {
91  /* Setup PLL for 100MHz and switch main system clocking */
92  Chip_Clock_SetupMainPLLHz(clkin, CGU_IRC_FREQ, 110 * 1000000, 110 * 1000000);
93  Chip_Clock_SetBaseClock(CLK_BASE_MX, CLKIN_MAINPLL, true, false);
94  }
95 
96  /* Setup PLL for maximum clock */
97  Chip_Clock_SetupMainPLLHz(clkin, OscRateIn, core_freq, core_freq);
98 
99  if (setbase) {
100  /* Setup system base clocks and initial states. This won't enable and
101  disable individual clocks, but sets up the base clock sources for
102  each individual peripheral clock. */
103  for (i = 0; i < (sizeof(InitClkStates) / sizeof(InitClkStates[0])); i++) {
104  Chip_Clock_SetBaseClock(InitClkStates[i].clk, InitClkStates[i].clkin,
105  InitClkStates[i].autoblock_enab, InitClkStates[i].powerdn);
106  }
107  }
108 }
