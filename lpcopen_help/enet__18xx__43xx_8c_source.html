63 /* Basic Ethernet interface initialization */
64 void Chip_ENET_Init(LPC_ENET_T *pENET)
65 {
66  Chip_Clock_EnableOpts(CLK_MX_ETHERNET, true, true, 1);
67 
68  reset(pENET);
69 
70  /* Setup MII link divider to /102 and PHY address 1 */
71  Chip_ENET_SetupMII(pENET, 4, 1);
72 
73  /* Enhanced descriptors, burst length = 1 */
74  pENET->DMA_BUS_MODE = DMA_BM_ATDS | DMA_BM_PBL(1) | DMA_BM_RPBL(1);
75 
76  /* Initial MAC configuration for checksum offload, full duplex,
77  100Mbps, disable receive own in half duplex, inter-frame gap
78  of 64-bits */
79  pENET->MAC_CONFIG = MAC_CFG_BL(0) | MAC_CFG_IPC | MAC_CFG_DM |
80  MAC_CFG_DO | MAC_CFG_FES | MAC_CFG_PS | MAC_CFG_IFG(3);
81 
82  /* Setup default filter */
83  pENET->MAC_FRAME_FILTER = MAC_FF_PR | MAC_FF_RA;
84 
85  /* Flush transmit FIFO */
86  pENET->DMA_OP_MODE = DMA_OM_FTF;
87 
88  /* Setup DMA to flush receive FIFOs at 32 bytes, service TX FIFOs at
89  64 bytes */
90  pENET->DMA_OP_MODE |= DMA_OM_RTC(1) | DMA_OM_TTC(0);
91 
92  /* Clear all MAC interrupts */
93  pENET->DMA_STAT = DMA_ST_ALL;
94 
95  /* Enable MAC interrupts */
96  pENET->DMA_INT_EN = 0;
97 }

99 /* Ethernet interface shutdown */
100 void Chip_ENET_DeInit(LPC_ENET_T *pENET)
101 {
102  /* Disable packet reception */
103  pENET->MAC_CONFIG = 0;
104 
105  /* Flush transmit FIFO */
106  pENET->DMA_OP_MODE = DMA_OM_FTF;
107 
108  /* Disable receive and transmit DMA processes */
109  pENET->DMA_OP_MODE = 0;
110 
111  Chip_Clock_Disable(CLK_MX_ETHERNET);
112 }

114 /* Sets up the PHY link clock divider and PHY address */
115 void Chip_ENET_SetupMII(LPC_ENET_T *pENET, uint32_t div, uint8_t addr)
116 {
117  /* Save clock divider and PHY address in MII address register */
118  phyCfg = MAC_MIIA_PA(addr) | MAC_MIIA_CR(div);
119 }

121 /* Starts a PHY write via the MII */
122 void Chip_ENET_StartMIIWrite(LPC_ENET_T *pENET, uint8_t reg, uint16_t data)
123 {
124  /* Write value at PHY address and register */
125  pENET->MAC_MII_ADDR = phyCfg | MAC_MIIA_GR(reg) | MAC_MIIA_W;
126  pENET->MAC_MII_DATA = (uint32_t) data;
127  pENET->MAC_MII_ADDR |= MAC_MIIA_GB;
128 }
