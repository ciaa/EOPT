84 /* Master transfer state change handler handler */
85 uint32_t Chip_I2CM_XferHandler(LPC_I2C_T *pI2C, I2CM_XFER_T *xfer)
86 {
87  uint32_t cclr = I2C_CON_FLAGS;
88 
89  switch (Chip_I2CM_GetCurState(pI2C)) {
90  case 0x08: /* Start condition on bus */
91  case 0x10: /* Repeated start condition */
92  pI2C->DAT = (xfer->slaveAddr << 1) | (xfer->txSz == 0);
93  break;
94 
95  /* Tx handling */
96  case 0x20: /* SLA+W sent NAK received */
97  case 0x30: /* DATA sent NAK received */
98  if ((xfer->options & I2CM_XFER_OPTION_IGNORE_NACK) == 0) {
99  xfer->status = I2CM_STATUS_NAK;
100  cclr &= ~I2C_CON_STO;
101  break;
102  }
103 
104  case 0x18: /* SLA+W sent and ACK received */
105  case 0x28: /* DATA sent and ACK received */
106  if (!xfer->txSz) {
107  if (xfer->rxSz) {
108  cclr &= ~I2C_CON_STA;
109  }
110  else {
111  xfer->status = I2CM_STATUS_OK;
112  cclr &= ~I2C_CON_STO;
113  }
114 
115  }
116  else {
117  pI2C->DAT = *xfer->txBuff++;
118  xfer->txSz--;
119  }
120  break;
121 
122  /* Rx handling */
123  case 0x58: /* Data Received and NACK sent */
124  case 0x50: /* Data Received and ACK sent */
125  *xfer->rxBuff++ = pI2C->DAT;
126  xfer->rxSz--;
127 
128  case 0x40: /* SLA+R sent and ACK received */
129  if ((xfer->rxSz > 1) || (xfer->options & I2CM_XFER_OPTION_LAST_RX_ACK)) {
130  cclr &= ~I2C_CON_AA;
131  }
132  if (xfer->rxSz == 0) {
133  xfer->status = I2CM_STATUS_OK;
134  cclr &= ~I2C_CON_STO;
135  }
136  break;
137 
138  /* NAK Handling */
139  case 0x48: /* SLA+R sent NAK received */
140  xfer->status = I2CM_STATUS_SLAVE_NAK;
141  cclr &= ~I2C_CON_STO;
142  break;
143 
144  case 0x38: /* Arbitration lost */
145  xfer->status = I2CM_STATUS_ARBLOST;
146  break;
147 
148  case 0x00: /* Bus Error */
149  xfer->status = I2CM_STATUS_BUS_ERROR;
150  cclr &= ~I2C_CON_STO;
151  break;
152  default:
153  xfer->status = I2CM_STATUS_ERROR;
154  cclr &= ~I2C_CON_STO;
155  break;
156  }
157 
158  /* Set clear control flags */
159  pI2C->CONSET = cclr ^ I2C_CON_FLAGS;
160  pI2C->CONCLR = cclr & ~I2C_CON_STO;
161 
162  return xfer->status != I2CM_STATUS_BUSY;
163 }

176 /* Transmit and Receive data in master mode */
177 uint32_t Chip_I2CM_XferBlocking(LPC_I2C_T *pI2C, I2CM_XFER_T *xfer)
178 {
179  uint32_t ret = 0;
180  /* start transfer */
181  Chip_I2CM_Xfer(pI2C, xfer);
182 
183  while (ret == 0) {
184  /* wait for status change interrupt */
185  while ( Chip_I2CM_StateChanged(pI2C) == 0) {}
186  /* call state change handler */
187  ret = Chip_I2CM_XferHandler(pI2C, xfer);
188  }
189  return ret;
190 }

192 /* Master tx only */
193 uint32_t Chip_I2CM_Write(LPC_I2C_T *pI2C, const uint8_t *buff, uint32_t len)
194 {
195  uint32_t txLen = 0, err = 0;
196 
197  /* clear state change interrupt status */
198  Chip_I2CM_ClearSI(pI2C);
199  /* generate START condition */
200  Chip_I2CM_SendStart(pI2C);
201 
202  while ((txLen < len) && (err == 0)) {
203  /* wait for status change interrupt */
204  while ( Chip_I2CM_StateChanged(pI2C) == 0) {}
205 
206  /* check status and send data */
207  switch (Chip_I2CM_GetCurState(pI2C)) {
208  case 0x08: /* Start condition on bus */
209  case 0x10: /* Repeated start condition */
210  case 0x18: /* SLA+W sent and ACK received */
211  case 0x28: /* DATA sent and ACK received */
212  Chip_I2CM_WriteByte(pI2C, buff[txLen++]);
213  break;
214 
215  case 0x38: /* Arbitration lost */
216  break;
217 
218  default: /* we shouldn't be in any other state */
219  err = 1;
220  break;
221  }
222  /* clear state change interrupt status */
223  Chip_I2CM_ClearSI(pI2C);
224  }
225 
226  return txLen;
227 }

229 /* Sequential master read */
230 uint32_t Chip_I2CM_Read(LPC_I2C_T *pI2C, uint8_t *buff, uint32_t len)
231 {
232  uint32_t rxLen = 0, err = 0;
233 
234  /* clear state change interrupt status */
235  Chip_I2CM_ClearSI(pI2C);
236  /* generate START condition and auto-ack data received */
237  pI2C->CONSET = I2C_CON_AA | I2C_CON_STA;
238 
239  while ((rxLen < len) && (err == 0)) {
240  /* wait for status change interrupt */
241  while ( Chip_I2CM_StateChanged(pI2C) == 0) {}
242 
243  /* check status and send data */
244  switch (Chip_I2CM_GetCurState(pI2C)) {
245  case 0x08: /* Start condition on bus */
246  case 0x10: /* Repeated start condition */
247  case 0x40: /* SLA+R sent and ACK received */
248  case 0x50: /* Data Received and ACK sent */
249  buff[rxLen++] = Chip_I2CM_ReadByte(pI2C);
250  break;
251 
252  case 0x38: /* Arbitration lost */
253  break;
254 
255  default: /* we shouldn't be in any other state */
256  err = 1;
257  break;
258  }
259  /* clear state change interrupt status */
260  Chip_I2CM_ClearSI(pI2C);
261  }
262 
263  return rxLen;
